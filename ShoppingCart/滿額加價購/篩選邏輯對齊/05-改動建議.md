var validSalepages = salepageList
    .Where(x => x.IsSoldOut == false)
    // ... SellingTime 過濾（Story #596792 已修）
    .ToList();

// 修改後
var validSalepages = salepageList
    .Where(x => x.IsSoldOut == false)
    // ... SellingTime 過濾（Story #596792 已修）

    .Where(x => x.IsClosed == false)            // 新增：排除商品頁關閉
    // IsShareToBuy 排除：待需求確認後補充
    .ToList();
