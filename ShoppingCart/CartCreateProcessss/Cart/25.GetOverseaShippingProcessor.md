取得國家地區配送清單


這邊 context.Data.ShippingAreaList  會有東西


```csharp
private async Task SetOverseaShippingArea(
    CartContext context,
    IEnumerable<ShoppingCartClientDeliveryTypeEntity> overseaShippings,
    List<ShippingAreaEntity> areaList)
{
    //// 是否有本地配送
    var hasLocalShipping = context.Data.CheckoutType.DeliveryTypeList.Except(overseaShippings, new ShoppingCartClientDeliveryTypeComparer()).Any();
    context.Data.CanOverseaShipping = areaList.Any() == true;
    context.Data.ShippingAreaList = new List<ShippingAreaEntity>();

    //// 有跨境配送地區
    if (areaList.Any())
    {
        var local = await this.GetLocalShippingArea(context.ShopId, false);

        if (hasLocalShipping && local != null)
        {
            areaList.Add(local);
        }

        areaList.ForEach(i => i.Name = Nine1.Cart.Common.Translations.Backend.Service.ShippingArea.GetString(i.LanguageKey));

        //// 同國家需要有特定排序時，按照ShippingAreaSort排序
        context.Data.ShippingAreaList = areaList.OrderBy(a => a.CountryEnglishName).ThenBy(a => a.ShippingAreaSort).ToList();

        return;
    }

    if (hasLocalShipping)
    {
        context.Data.ShippingAreaList = new List<ShippingAreaEntity>();
        return;
    }

    //// 無跨境配送地區且商品僅能跨境配送時拋出例外
    throw new CalculateException(CalculateExceptionTypeEnum.OverseaShippingNotAvailable);
}
```
