計算購物車可用結帳付款運送方式


## 重大1 (isMatchDeliveryArea = False => IsCheapestFee = false)

```csharp
var filteredDeliveryList = shoppingCart.CheckoutType.DeliveryTypeList?.Where(x =>
{
    //// 配送地區與宅配重量計價配送方式沒有符合的配送方式不列入計算
    if (x.IsDomesticWeightPricing() && x.isMatchDeliveryArea == false)
    {
        return false;
    }

    return true;
}).ToList();

//// 依據物流方式分組
var deliveryTypeGroupList = filteredDeliveryList?.GroupBy(i => new { i.ShippingProfileTypeDef, i.ShippingAreaId });

//// 清除前一次計算結果
shoppingCart.CheckoutType.DeliveryTypeList.ForEach(x =>
{
    x.IsCheapestFee = false;
});

//// 計算物流運費
foreach (var deliveryTypeGroup in deliveryTypeGroupList)
{
    //// 依照運費排序後，再依照溫層別分組
    var temperatureTypeGroupList = deliveryTypeGroup.OrderBy(i => i.TotalFee).GroupBy(i => i.TemperatureTypeDef);

    //// 有使用運費券
    if (this.IsSelectedFeeCoupon(shoppingCart) == true)
    {
        temperatureTypeGroupList = deliveryTypeGroup.OrderBy(i => i.EstimateTotalFee.HasValue == true ? i.EstimateTotalFee : i.TotalFee)
                                                    .ThenBy(i => i.Fee)
                                                    .GroupBy(i => i.TemperatureTypeDef);
    }

    foreach (var temperatureTypeGroup in temperatureTypeGroupList)
    {
        var cheapest = temperatureTypeGroup.FirstOrDefault();

        if (cheapest != null)
        {
            cheapest.IsCheapestFee = true;
        }
    }
}
```


## 重大2 (CheckoutType.DeliveryTypeList => CheckoutType.DisplayShippingTypeList)

## shoppingCart.CheckoutType.DisplayShippingTypeList 設定


```csharp
//// 依據物流方式與配送區域序號分組
var deliveryTypeGroupList = shoppingCart.CheckoutType.DeliveryTypeList
                                        .Where(delivery => payShippingMappingList.Select(p => p.ShippingProfileTypeDef)
                                                                                    .Contains(delivery.ShippingProfileTypeDef))
                                        .GroupBy(i => new { i.ShippingProfileTypeDef, i.ShippingAreaId, i.Sort })
                                        .ToList();

//// 設定顯示用物流資訊
var shippingTypeGroup = deliveryTypeGroupList
                        .OrderBy(i => i.Key.Sort)
                        .Select(i => new CheckoutShippingTypeGroupEntity
                        {
                            ShippingProfileTypeDef = i.Key.ShippingProfileTypeDef,
                            ShippingAreaId = i.Key.ShippingAreaId,
                            DeliveryTypeList = i.ToList(),
                            CanDefaultSelected = i.First().CanDefaultSelected,
                            IsRecommend = canDefaultSelectedCheapest && i.Key.ShippingProfileTypeDef == cheapestFeeShippingProfile.ShippingProfileTypeDef &&
                                            i.Key.ShippingAreaId == cheapestFeeShippingProfile.ShippingAreaId
                        }).ToList();

shoppingCart.CheckoutType.DisplayShippingTypeList = shippingTypeGroup;
```
