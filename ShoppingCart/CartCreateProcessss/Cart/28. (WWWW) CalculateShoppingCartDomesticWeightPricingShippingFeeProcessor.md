## 非本地配送，不計算宅配重量計價

 ## 已選地區是否合法

 bool isSelectedShippingAreaValid = await this._domesticWeightPricingShippingService.IsShippingAreaValidAsync(context);


 ==> No shipping area was selected or the shipping area not sub shipping area. Skip this process.


## 檢查是否所有商品都有重量

 bool isAllSalepageWeightHasValue = context.Data.IsAllSalePagesWeightHasValue();


##  取得購物車宅配重量計價配送方式



var checkoutDeliveryTypes = context.Data.CheckoutType?.DeliveryTypeList?.Where(item => item.IsDomesticWeightPricing() == true);


==> No matched domestics weight pricing delivery type. Skip this process.


## 這邊用到 SubShippingAreaId

var selectedShippingAreaId = context.Data.SelectedShippingArea?.SubShippingAreaId;


## 查運費表


```csharp
//// 取得運費表
var shopShippingWeightProfile = await this._shopShippingWeightProfileService.GetShopShippingWeightProfileAsync(context.ShopId, deliveryType.Id);
var shippingAreaWeightFee = shopShippingWeightProfile.ShippingAreaList.FirstOrDefault(x => x.ShippingAreaId == selectedShippingAreaId);

//// 沒有符合的配送區域的配送方式在P2無法顯示運費，前端需要改顯示待計算。
deliveryType.isMatchDeliveryArea = shippingAreaWeightFee != null;

if (shippingAreaWeightFee == null)
{
    this._logger.LogWarning($"Can't find the shipping area from delivery type - ShopId: {context.ShopId}, DeliveryTypeId: {deliveryType.Id}, ShippingAreaId: {selectedShippingAreaId}");
    continue;
}
```

## 超重


```csharp
if (context.Data.TotalWeight > shippingAreaWeightFee.MaxWeight)
{
    this._logger.LogInformation($"Shopping cart total weight exceed the maximum weight of shipping area");
    continue;
}
```

## 設定運費


```csharp
//// 需要將運費類型設定回原始狀態
deliveryType.FeeTypeDef = deliveryType.OriginalFeeTypeDef;

var fee = this.GetShippingWeightFee(shippingAreaWeightFee, context.Data);
deliveryType.OverPrice = shippingAreaWeightFee.OverPriceThreshold;
deliveryType.Fee = fee;
deliveryType.OriginalFee = fee;
```
