## 金流 檢查選取的付款方式是否還在付款顯示清單中，若不在，將選取的付款方式清空

```csharp
if (context.SelectedCheckoutPayTypeGroup == null)
{
    return;
}

var isExist = false;
var statisticsTypeDefEnum = (StatisticsTypeDefEnum)Enum.Parse(typeof(StatisticsTypeDefEnum), context.SelectedCheckoutPayTypeGroup.StatisticsTypeDef);

if (NormalPayTypeList.Contains(statisticsTypeDefEnum) || (Enum.IsDefined(typeof(StatisticsTypeDefEnum), statisticsTypeDefEnum) && StatisticsTypeDefEnum.CommonThirdPartyPayType.HasFlag(statisticsTypeDefEnum)))
{
    //// 一般金流檢查群組類型是否有在顯示清單中
    isExist = context.CheckoutType.DisplayPayTypeList.Any(i => i.StatisticsTypeDef == statisticsTypeDefEnum.ToString());

    _logger.LogInformation("一般金流檢查群組類型是否有在顯示清單中:" + isExist.ToString());
}
else if (HasInstallmentPayTypeList.Contains(statisticsTypeDefEnum))
{
    //// 當金流為信用卡分期時，需檢查分期序號是否有在顯示清單中
    isExist = context.CheckoutType.DisplayPayTypeList.Any(i => i.StatisticsTypeDef == statisticsTypeDefEnum.ToString()
                                                            && i.InstallmentType.Id == context.SelectedCheckoutPayTypeGroup.InstallmentType.Id);

    _logger.LogInformation("當金流為信用卡分期時，需檢查分期序號是否有在顯示清單中:" + isExist.ToString());
}

//// 不存在時，將已選擇的結帳金流方式清空
if (isExist == false)
{
    context.SelectedCheckoutPayTypeGroup = null;

    _logger.LogInformation("不存在時，將已選擇的結帳金流方式清空:" + isExist.ToString());
}
```


##　調整預選的物流方式

```csharp
//// 以下兩種情況，需要指定預選的物流類型
//// 1. 無預選物流類型
//// 2. 海外物流
var noPreselectDeliveryTypeList =
    shoppingCart.CheckoutType
                .DisplayShippingTypeList
                .Where(x => x.DeliveryTypeList.Any(d => d.IsSelected) == false ||
                            x.ShippingProfileTypeDef == ShippingProfileTypeDefEnum.Oversea.ToString());

foreach (var displayShippingType in noPreselectDeliveryTypeList)
{
    if (shoppingCart.SelectedCheckoutShippingTypeGroup != null &&
        displayShippingType.DeliveryTypeList.Any(d => d.IsSelected))
    {
        _logger.LogInformation("依配送地區決定預選_已選");
        continue;
    }

    //// 依配送地區決定預選
    if (displayShippingType.ShippingProfileTypeDef == ShippingProfileTypeDefEnum.Oversea.ToString() &&
        shoppingCart.SelectedShippingArea != null &&
        shoppingCart.SelectedShippingArea.IsLocal == false)
    {
        _logger.LogInformation("依配送地區決定預選_海外");
        displayShippingType.DeliveryTypeList.ForEach(i => i.IsSelected = i.Id == shoppingCart.SelectedShippingArea.DeliveryTypeId);
    }
    else
    {
        _logger.LogInformation("依配送地區決定預選_最便宜");
        displayShippingType.DeliveryTypeList.ForEach(i => i.IsSelected = i.IsCheapestFee);
    }
}
```



## 預選結帳運送方式

```csharp
/// <summary>
/// 預選結帳運送方式
/// </summary>
/// <param name="shoppingCart">購物車資料</param>
private void PickupSelectedShippingType(
    CartEntity shoppingCart)
{
    var selectedCheckoutShippingType = shoppingCart.SelectedCheckoutShippingTypeGroup;
    var displayShippingTypeList = shoppingCart.CheckoutType.DisplayShippingTypeList;
    var originShippingProfileTypeDef = selectedCheckoutShippingType?.ShippingProfileTypeDef ?? string.Empty;
    var originShippingAreaId = selectedCheckoutShippingType?.ShippingAreaId ?? 0;
    var originAddressBookId = selectedCheckoutShippingType?.AddressBookId ?? 0;

    //// 調整預選的物流方式
    //// 本地選取最便宜, 海外依配送地區選取
    this.ArrangePreselect(shoppingCart);

    if (selectedCheckoutShippingType != null)
    {
        _logger.LogInformation("不需要預選結帳運送方式，需要重新取得計算過的運送方式");
        //// 不需要預選結帳運送方式，需要重新取得計算過的運送方式
        selectedCheckoutShippingType =
            displayShippingTypeList.FirstOrDefault(i => i.ShippingProfileTypeDef == selectedCheckoutShippingType.ShippingProfileTypeDef &&
                                                        i.ShippingAreaId == selectedCheckoutShippingType.ShippingAreaId);
    }

    if (selectedCheckoutShippingType == null)
    {
        //// 沒有已選擇的物流方式，或是已選擇的物流方式不在可U用清單中，重新取得預選物流
        if (displayShippingTypeList.Count == 1)
        {
            _logger.LogInformation("只有一種結帳運送方式，直接預選此種結帳方式");
            //// 只有一種結帳運送方式，直接預選此種結帳方式
            selectedCheckoutShippingType = displayShippingTypeList.FirstOrDefault(i => i.DeliveryTypeList.Any(d => d.IsNormalSalepageOverweight == false));
        }
        else if (displayShippingTypeList.Any(i => i.IsRecommend && i.DeliveryTypeList.Any(d => d.IsNormalSalepageOverweight == false)))
        {
            _logger.LogInformation("有多種結帳運送方式，預選建議運送方式");
            //// 有多種結帳運送方式，預選建議運送方式
            selectedCheckoutShippingType = displayShippingTypeList.First(i => i.IsRecommend && i.DeliveryTypeList.Any(d => d.IsNormalSalepageOverweight == false));
        }
        else
        {
            if (shoppingCart.CanOverseaShipping == false ||
                shoppingCart.SelectedShippingArea == null ||
                shoppingCart.SelectedShippingArea.IsLocal)
            {
                _logger.LogInformation("無推薦的運送方式，排除不可預選運送方式後，使用最便宜的運送方式");
                //// 無推薦的運送方式，排除不可預選運送方式後，使用最便宜的運送方式
                selectedCheckoutShippingType = shoppingCart.CheckoutType.DisplayShippingTypeList
                                                            .OrderBy(i =>
                                                                    {
                                                                        var estimateTotalFee = i.DeliveryTypeList.Where(ii => ii.IsSelected && ii.EstimateTotalFee.HasValue)
                                                                                                .Sum(ii => ii.EstimateTotalFee);

                                                                        return estimateTotalFee == default ? estimateTotalFee : i.TotalFee;
                                                                    })
                                                            .FirstOrDefault(i => i.CanDefaultSelected && i.DeliveryTypeList.Any(d => d.IsNormalSalepageOverweight == false));
            }
            else
            {
                _logger.LogInformation("依選擇配送地區的物流方式，進行運送方式的預選");
                //// 依選擇配送地區的物流方式，進行運送方式的預選
                selectedCheckoutShippingType = shoppingCart.CheckoutType.DisplayShippingTypeList
                                                            .FirstOrDefault(x => x.DeliveryTypeList.Any(y => y.Id == shoppingCart.SelectedShippingArea.DeliveryTypeId && y.IsNormalSalepageOverweight == false));
            }
        }
    }

    //// 若運送方式不變，帶入原本選擇的收件人
    if (originAddressBookId > 0 && selectedCheckoutShippingType != null
        && originShippingProfileTypeDef == selectedCheckoutShippingType.ShippingProfileTypeDef
        && originShippingAreaId == selectedCheckoutShippingType.ShippingAreaId)
    {
        _logger.LogInformation("若運送方式不變，帶入原本選擇的收件人");
        selectedCheckoutShippingType.AddressBookId = originAddressBookId;
    }

    shoppingCart.SelectedCheckoutShippingTypeGroup = selectedCheckoutShippingType;
}
```
