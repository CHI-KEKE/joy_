foreach (var deliveryType in context.Data.CheckoutType.DeliveryTypeList)
{
    if (deliveryType.IsDomesticWeightPricing() == false)
    {
        continue;
    }

    decimal maxWeight = await this._shopShippingWeightProfileService.GetShopShippingWeightProfileMaxWeightAsync(context.Data.ShopId, deliveryType.Id);
    deliveryType.WeightLimit = (int)(maxWeight * 1000);
}
