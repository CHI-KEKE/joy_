var cartEntity = context.Data;
var isLocalShipping = cartEntity.SelectedShippingArea is null ||
                        cartEntity.SelectedShippingArea.IsLocal;

if (isLocalShipping && cartEntity.CheckoutType.DeliveryTypeList is not null)
{
    //// 不能跨境配送則清除跨境配送方式
    cartEntity.CheckoutType.DeliveryTypeList =
        cartEntity.CheckoutType.DeliveryTypeList.Where(s => s.ShippingProfileTypeDef != nameof(ShippingProfileTypeDefEnum.Oversea))
                    .ToList();

    _logger.LogInformation($"清除海外配送，清除後的DeliveryTypeId : {cartEntity.CheckoutType.DeliveryTypeList.Select(x => x.Id).ToJson()}");

    return Task.CompletedTask;
}

if (isLocalShipping == false && cartEntity.CheckoutType.DeliveryTypeList is not null)
{
    //// 配送區域非本地 => 他只能是海外配送 或是 門市自取(海外)，移除不合理的配送方式
    cartEntity.CheckoutType.DeliveryTypeList =
        cartEntity.CheckoutType.DeliveryTypeList.Where(
                        s => s.ShippingProfileTypeDef == nameof(ShippingProfileTypeDefEnum.Oversea) ||
                            (cartEntity.HasAbroadStorePickup && s.ShippingProfileTypeDef ==
                                nameof(ShippingProfileTypeDefEnum.LocationPickup)))
                    .ToList();

    _logger.LogInformation($"清除非海外配送，清除後的DeliveryTypeId : {cartEntity.CheckoutType.DeliveryTypeList.Select(x => x.Id).ToJson()}");
}

if (context.Data.SalePageGiftCouponGroupList.Any() == false)
{
    var hasPayTypeList = context.Data.CheckoutType.PayTypeList?.Any() ?? false;
    var hasDeliveryTypeList = context.Data.CheckoutType.DeliveryTypeList?.Any() ?? false;
    if (hasPayTypeList == false || hasDeliveryTypeList == false)
    {
        //// 砍完之後，金流或物流為空，中斷流程
        _logger.LogInformation($"金流或物流為空，中斷流程，{context.Data.CartUniqueKey}");
        //context.IsInterrupted = true;
    }
}

return Task.CompletedTask;
