```csharp
//// 指定金流對應的物流
//// 此處無法判斷真超重、假超重，需經過後面促購計算才能確定，故不過濾超重物流，如最後預選出來為真超重，在P2由使用者重新選擇。
var promoCodeDisplayShippingTypes = context.Data.CheckoutType.DeliveryTypeList.Where(i => promoCodePayShippingMappings.Select(i => i.ShippingProfileTypeDef).Contains(i.ShippingProfileTypeDef))
                                                                                .GroupBy(i => new { i.ShippingProfileTypeDef, i.ShippingAreaId, i.Sort })
                                                                                .OrderBy(i => i.Where(d => d.IsCheapestFee).Sum(d => d.TotalFee))
                                                                                .ThenBy(i => i.Key.Sort)
                                                                                .Select(i => new CheckoutShippingTypeGroupEntity
                                                                                {
                                                                                    ShippingProfileTypeDef = i.Key.ShippingProfileTypeDef!,
                                                                                    ShippingAreaId = i.Key.ShippingAreaId,
                                                                                    DeliveryTypeList = i.ToList(),
                                                                                    CanDefaultSelected = i.First().CanDefaultSelected,
                                                                                    IsRecommend = i.First().CanDefaultSelected && (i.Key.ShippingProfileTypeDef == i.First().ShippingProfileTypeDef &&
                                                                                    i.Key.ShippingAreaId == i.First().ShippingAreaId)
                                                                                }).ToList();
```
