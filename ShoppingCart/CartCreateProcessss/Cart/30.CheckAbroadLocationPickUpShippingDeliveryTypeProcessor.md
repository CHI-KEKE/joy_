確認境外門市的可配送方式

```csharp
CartEntity shoppingCart = context.Data;
ShippingAreaEntity selectedShippingArea = shoppingCart.SelectedShippingArea;

if (selectedShippingArea == null)
{
    return Task.CompletedTask;
}

var selectedLocationPickupDeliveryTypeList = shoppingCart.CheckoutType.DeliveryTypeList
    .Where(x => x.Id == selectedShippingArea.DeliveryTypeId && x.IsLocationPickup() == true).ToList();

var isAbroadLocationPickUpDelivery = selectedLocationPickupDeliveryTypeList.Any() &&
                                        selectedShippingArea.IsLocal == false;

//// 如果配送地區是境外門市自取 則將可配送區域設定為境外門市自取
if (isAbroadLocationPickUpDelivery)
{
    shoppingCart.CheckoutType.DeliveryTypeList = selectedLocationPickupDeliveryTypeList;
}

return Task.CompletedTask;
```
