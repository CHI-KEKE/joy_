## IsApplePaySupported / GoolgePay

```csharp
if (this.IsProduction &&
    userClientTrack.TrackSourceType != UserClientTrackSourceTypeEnum.AndroidApp)
{
    avaliablePayProfileTypeDefEnum = avaliablePayProfileTypeDefEnum ^ PayProfileTypeDefEnum.GooglePay;
}
else if (market == "HK" &&
    this.IsProduction &&
    userClientTrack.CheckAppVersionIsUnsupported(new Version(24, 13)) == true)
{
    //// HK 24.13.0 後才支援 GooglePay
    avaliablePayProfileTypeDefEnum = avaliablePayProfileTypeDefEnum ^ PayProfileTypeDefEnum.GooglePay;
}

//// 取得 Apple Pay PSP 設定
var applePayPspSetting = await GetApplePayPspSettingAsync(context.ShopId, userClientTrack);

//// 根據 PSP 類型和裝置判斷是否支援 Apple Pay
if (!IsApplePaySupported(applePayPspSetting, userClientTrack))
{
    _logger.LogInformation("要移除 ApplePay");
    avaliablePayProfileTypeDefEnum = avaliablePayProfileTypeDefEnum ^ PayProfileTypeDefEnum.ApplePay;
}
else if (market == "HK" &&
    this.IsProduction &&
    userClientTrack.CheckAppVersionIsUnsupported(new Version(24, 13)) == true)
{
    //// HK 24.13.0 後才支援 ApplePay
    avaliablePayProfileTypeDefEnum = avaliablePayProfileTypeDefEnum ^ PayProfileTypeDefEnum.ApplePay;
}
```
