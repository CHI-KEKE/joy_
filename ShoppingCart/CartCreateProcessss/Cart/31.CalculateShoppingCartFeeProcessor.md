```csharp
foreach (var deliveryTypeItem in context.Data.CheckoutType.DeliveryTypeList)
{
    var shopShippingTypeId = deliveryTypeItem.Id;
    var selectedShippingAreaId = context.Data?.CalculateShippingAreaId;
    var shopShippingTypePromotions = context.Data!.FeePromotionInfoList?.Where(x => x.ShopShippingTypeId == shopShippingTypeId && x.SourceType == SourceTypeEnum.FeePromotion).ToList();
    var matchedShippingAreaPromotion = shopShippingTypePromotions.FirstOrDefault(x => x.ConditionList.Any(y => y.ShippingAreaId == selectedShippingAreaId));

    var matchedPromotionCondition = matchedShippingAreaPromotion?.ConditionList?.FirstOrDefault(x => x.IsMatched);
    if (matchedPromotionCondition != null)
    {
        //// 顯示運費活動節點
        context.Data.FeeShippingDiscountType = nameof(SourceTypeEnum.Promotion);
        context.Data.FeeShippingDiscount = matchedShippingAreaPromotion!.DiscountAmount;

        //// 有中運費活動，套用活動設定
        this.ApplyShopShippingTypePromotionFee(deliveryTypeItem, matchedShippingAreaPromotion!, matchedPromotionCondition);
        if (matchedPromotionCondition.IsFree == true)
        {
            context.Data.FeeShippingDiscountType += "FreeShipping";
        }
    }
    else
    {
        if (shopShippingTypePromotions.Any())
        {
            //// 有設運費活動但是沒中或未符合，類型改回 重量計費/固定運費
            //// (即使活動的配送地區不是這次結帳所選地區，也就是說不會再套用運費表上的滿額免運)
            //// VSTS144008 Revised: 如果原始計費方式為"重量計費"，則將計費方式改回 "重量計費"；若為其它計費方式則改回"固定運費"
            deliveryTypeItem.FeeTypeDef = deliveryTypeItem.OriginalFeeTypeDef == nameof(ShippingFeeTypeDefEnum.WeightBilling) ? nameof(ShippingFeeTypeDefEnum.WeightBilling) : nameof(ShippingFeeTypeDefEnum.Fixed);
        }

        //// 沒中活動，套用配送方式原本的設定
        this.ApplyShopShippingTypeFee(salepageGroupList, deliveryTypeItem);
    }
}
```
