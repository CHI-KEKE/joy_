計算運費券流程

```csharp
public async Task ExecuteAsync(CartContext context)
{
    //// 避免金物流計算過程殘留券的資料，設定折扣前先重新清除一次
    this.ResetFeeShippingCouponDiscount(context);

    //// 運費券僅能選擇一張
    var feeShippingCoupon = context.Data.CouponInfoList.Where(x => x is { Type: "Shipping", IsSelected: true }).SingleOrDefault();

    if (feeShippingCoupon == null)
    {
        return;
    }

    if (context.Data.SelectedCheckoutShippingTypeGroup == null)
    {
        this._logger.LogInformation("運費券，沒有預選物流方法");
        return;
    }

    //// 若以達到免運，則不預選運費券
    if (context.Data.SelectedCheckoutShippingTypeGroup.TotalFee == 0 && context.Data.IsCouponPreSelect == true)
    {
        feeShippingCoupon.IsSelected = false;
        return;
    }

    //// 檢查物流項目序號是否存在
    if (feeShippingCoupon!.FeeItemIds == null || feeShippingCoupon!.FeeItemIds.Any() == false)
    {
        this._logger.LogError($"物流項目不存在：{context.Data.SelectedCheckoutShippingTypeGroup}");
        return;
    }

    var isSelectedFeeForFree = context.Data.SelectedCheckoutShippingTypeGroup!.DeliveryTypeList.Where(x => x.IsSelected == true).All(x => x.TotalFee == 0);
    //// 如果是目前選擇運費沒有免運，或是購物車在P1頁
    if (isSelectedFeeForFree == false || context.Data.IsCartCalculate == true)
    {
        this.SetFeeShippingCouponDiscountType(context, feeShippingCoupon);
    }

    var calculateFeeDetailSummary = await this._couponService.GetCalculateEstimateAllFeeDiscountAsync(context, feeShippingCoupon);

    //// 計算選取的物流方式折抵
    foreach (var deliveryType in context.Data.SelectedCheckoutShippingTypeGroup!.DeliveryTypeList)
    {
        var feeCouponDetailList = calculateFeeDetailSummary.Where(x => x.FeeItemId == deliveryType.Id).ToList();
        var couponDiscount = feeCouponDetailList.Sum(x => x.Amount);

        //// 有符合的物流序號，且有運費攤提
        if (feeCouponDetailList.Any() == true && couponDiscount > 0)
        {
            //// 物流方式折扣加總
            deliveryType.ECouponDiscount = couponDiscount;
            deliveryType.ECouponId = feeShippingCoupon.CouponId;
            deliveryType.ECouponSlaveId = feeShippingCoupon.CouponSlaveId ?? 0;
            //// 為了 P2 顯示實際運費被折抵多少金額，Fee 需顯示原配送方式運費金額
            deliveryType.Fee = deliveryType.OriginalFee;
            deliveryType.TotalFee = deliveryType.OriginalFee - couponDiscount;
            deliveryType.IsReachFreeFee = deliveryType.TotalFee == 0; //// 券折扣把運費折到變0，就有可能變成免運
        }
    }

    //// 總運費券折抵
    context.Data.FeeShippingDiscount = context.Data.SelectedCheckoutShippingTypeGroup.DeliveryTypeList.Where(x => x.IsSelected == true).Sum(x => x.ECouponDiscount);

    this._couponService.SetFeeCouponCalculateEstimateDeliveryFee(context.Data.CheckoutType.DeliveryTypeList, calculateFeeDetailSummary);
}
```
