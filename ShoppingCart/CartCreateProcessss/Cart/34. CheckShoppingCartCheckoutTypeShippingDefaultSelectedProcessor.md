檢查購物車物流預選(含設定是否可預選)

```csharp
//// 設定預選預設值
context.Data.CheckoutType.DeliveryTypeList.ForEach(i => i.CanDefaultSelected = true);

//// 設定宅配非預選區域為非預選
var list = await this._shippingAreaService.GetShippingAreaList();
var noDefaultSelectedShippingAreaList = list.Where(i => i.DefaultSelectedEnable == false);
context.Data.CheckoutType.DeliveryTypeList
    .Where(i => i.ShippingProfileTypeDef == ShippingProfileTypeDefEnum.Home.ToString())
    .ToList()
    .ForEach(i =>
    {
        i.CanDefaultSelected = noDefaultSelectedShippingAreaList.Any(x => x.Id == i.ShippingAreaId) ? false : true;
    });

//// 設定排除門市自取為非預選
var supportLocationPickupCanDefaultSelectedExclusionSetting = this._configurationService.GetConfiguration<string>(Common.Utils.Config.ConfigType.Normal, "LocationPickup:CanDefaultSelectedExclusion.Enabled.ShopIds");
var isLocationPickupCanDefaultSelectedExclusion = this.GetIsEnabledInSettingByShopId(context.ShopId, supportLocationPickupCanDefaultSelectedExclusionSetting);

if (isLocationPickupCanDefaultSelectedExclusion)
{
    context.Data.CheckoutType.DeliveryTypeList
        .Where(i => i.ShippingProfileTypeDef == ShippingProfileTypeDefEnum.LocationPickup.ToString())
        .ToList()
        .ForEach(i => i.CanDefaultSelected = false);
}
```
