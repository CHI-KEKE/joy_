var shoppingCart = context.Data;

var selectedShippingArea = shoppingCart.CheckoutType.DeliveryTypeList
                                        .FirstOrDefault(x => x.Id == shoppingCart.SelectedShippingArea?.DeliveryTypeId);

var isOversea = shoppingCart.CanOverseaShipping &&
                shoppingCart.SelectedShippingArea != null &&
                selectedShippingArea != null &&
                selectedShippingArea.ShippingProfileTypeDef != nameof(ShippingProfileTypeDefEnum.LocationPickup) &&
                shoppingCart.SelectedShippingArea.IsLocal == false;

if (isOversea == false)
{
    return;
}
