取得金物流上限訊息文字


```csharp
//// 當有任一筆金額上限 > 0，表示有被移除的金流或物流，需顯示金額最大的訊息文字
if (context.Data.CheckoutType.MaxOverLimitPayTypePrice > 0 || context.Data.CheckoutType.MaxOverLimitShippingTypePrice > 0)
{
    var maxOverLimitPrice = Math.Max(context.Data.CheckoutType.MaxOverLimitPayTypePrice, context.Data.CheckoutType.MaxOverLimitShippingTypePrice);

    context.Data.CheckoutType.DisplayMessage = string.Format(
        GetShoppingCartPayShippingPriceLimitDisplayMessage.MaxOverLimitPayTypePriceDisplayMessage,
        $"${maxOverLimitPrice}");
}

//// 當有任一筆金額下限 > 0，表示有被移除的金流或物流，需顯示訊息文字
if (context.Data.CheckoutType.MinUnderLimitPayTypePrice > 0)
{
    context.Data.CheckoutType.DisplayMessage =
        string.Format(
            GetShoppingCartPayShippingPriceLimitDisplayMessage.MinUnderLimitPayTypePriceDisplayMessage,
            $"${context.Data.CheckoutType.MinUnderLimitPayTypePrice}");
```
