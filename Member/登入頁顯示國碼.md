

## 結論

來自銷售市場的判斷

資料來源：資料庫表格 ShopSalesMarketSettings，該商店的 SalesMarket = "SG"
後端查詢：MarketModuleAgentService 查詢並快取結果
Filter 設定：DetectOfficialAttribute 將結果存入 ViewBag.SalesMarketSettings
視圖注入：InitialReactSetting.cshtml 將資料序列化成 JSON 並注入到 window.nineyi.MWeb.SalesMarketSettings
前端讀取：salesMarketSettingsProvider 從 window.nineyi 讀取並導出
元件使用：confirmCellPhone.tsx import Provider 並使用 SalesMarket 值
匹配國家：在 API 回傳的國家清單中找到 AliasCode = "SG" 的項目（CountryCode = "65"）
UI 顯示：將 label 設為 "SG+65" 並渲染到畫面

```typescript
export const getGlobalVariable = <T>(path: string): T => {
    // path = "MWeb.SalesMarketSettings"
    
    const paths = path.split('.');  // ["MWeb", "SalesMarketSettings"]
    let value = window.nineyi;      // 從 window.nineyi 開始
    
    for (const key of paths) {
        value = value[key];         // 逐層取得屬性
    }
    
    return value as T;
    // 最終回傳：window.nineyi.MWeb.SalesMarketSettings
};
```

```typescript
const ConfirmCellPhoneComponent: FC<IProps> = (props) => {
    const shopId = getGlobalVariable<number>('MWeb.ShopId');
    const [country, setCountry] = useState<ICountryProfileEntitySelector>(taiwanProfile);
    
    useEffect(() => {
        // ==========================================
        // 步驟 1: 從 API 取得該商店支援的國家清單
        // ==========================================
        fetchCountryProfileListByShopId(shopId).then((result) => {
            // API 呼叫：GET /api/v2/CountryProfile/GetCountryProfileListByShopId?shopId=12345
            // 回傳：
            // {
            //   "Data": [
            //     { "AliasCode": "SG", "CountryCode": "65", "Name": "新加坡", ... },
            //     { "AliasCode": "MY", "CountryCode": "60", "Name": "馬來西亞", ... },
            //     { "AliasCode": "TW", "CountryCode": "886", "Name": "台灣", ... }

            // }

            // ==========================================
            const initDefaultCountryAliasCode: string = 
                salesMarketSettings.SalesMarket || taiwanProfile.AliasCode;
            // salesMarketSettings.SalesMarket = "SG"

            //   "Id": 5,
```
