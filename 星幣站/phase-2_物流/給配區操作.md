/// <summary>
/// 取得 CheckoutReceiver
/// </summary>
/// <param name="shopId">商店序號</param>
/// <param name="shippingAreaId">shippingAreaId</param>
/// <returns></returns>
public async Task<List<CheckoutReceiverEntity>> GetCheckoutReceiverWithShippingAreaAsync(long shopId, long shippingAreaId)
{
    var memberLocationList = await this.GetMemberLocationListWithShippingAreaAsync(shopId, shippingAreaId);
    var receiverList = await AssignReceiverInfo(memberLocationList);
    
    return receiverList;
}
