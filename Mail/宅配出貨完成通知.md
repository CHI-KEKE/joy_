```sql
	use ERPDB
	
DECLARE @EndDateTime datetime = getdate();
DECLARE @StartDateTime datetime= DATEADD(hour, -1, @EndDateTime);

	SELECT ShippingOrderSlave_TradesOrderSlaveId
	FROM [ERPDB].[dbo].[ShippingOrderSlave] WITH(NOLOCK)
	WHERE [ShippingOrderSlave_ValidFlag] = 1 
	  AND [ShippingOrderSlave_StatusDef] ='Finish' 	  
	  AND [ShippingOrderSlave_StatusUpdatedDateTime] > @StartDateTime
	  AND [ShippingOrderSlave_StatusUpdatedDateTime] < @EndDateTime

	--取得宅配出貨完成 銷售子單
	SELECT ROW_NUMBER() OVER (ORDER BY [OrderSlaveFlow_TradesOrderCode]) AS Email_Num,
	       [OrderSlaveFlow_TradesOrderCode]
	from OrderSlaveFlow(nolock)
	WHERE [OrderSlaveFlow_ShippingProfileTypeDef] = 'Home'
	AND [OrderSlaveFlow_FulfillmentCode] = ''
	and OrderSlaveFlow_TradesOrderSlaveId = 24762
	GROUP BY [OrderSlaveFlow_TradesOrderCode]
```
