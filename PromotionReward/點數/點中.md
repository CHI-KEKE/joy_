USE ERPDB

SELECT DISTINCT ShopContract_ShopId, ContractFeature_Code, ShopContractFeatureInfo_InfoValue
FROM dbo.ContractFeature
INNER JOIN dbo.ShopContractFeature
	ON ContractFeature_Id = ShopContractFeature_ContractFeatureId
INNER JOIN dbo.ShopContractFeatureInfo
	ON ShopContractFeatureInfo_ShopContractFeatureId = ShopContractFeature_Id
INNER JOIN dbo.ShopContract
	ON ShopContractFeature_ShopContractId = ShopContract_Id
	AND ShopContractFeature_ShopId = ShopContract_ShopId
INNER JOIN dbo.Shop
	ON Shop_Id = ShopContract_ShopId
WHERE ContractFeature_ValidFlag = 1
	AND ShopContractFeature_ValidFlag = 1
	AND ShopContractFeatureInfo_ValidFlag = 1
	AND ShopContract_ValidFlag = 1
	AND ShopContract_IsOngoing = 1
	AND Shop_IsCRM = 1
	AND Shop_ValidFlag = 1
	AND ContractFeature_Code IN (
		'CRMTrial','CRMLite','CRM'
	)
	AND ShopContractFeatureInfo_InfoPropertyDef = 'MemberTierSourceTypeDef'
	AND ShopContractFeatureInfo_InfoValue = 'ThirdPartyAPI'
