

## 註冊

```csharp
//// 註冊系統多語系服務
builder.Services.ResgisterTranslationsClient(builder.Configuration);

/// <summary>
/// 註冊 TranslationsClient
/// </summary>
/// <param name="services">The <see cref="IServiceCollection"/> to add the service to.</param>
/// <param name="configuration">IConfiguration</param>
public static IServiceCollection ResgisterTranslationsClient(this IServiceCollection services, IConfiguration configuration)
{
    //// 若註冊這段壞掉 先記 log 不影響程式啟動
    try
    {
        var serviceBaseUrl = configuration["_N1CONFIG:TranslationSetting:ServiceBaseUrl"];
        var s3RestApiBaseUrl = configuration["_N1CONFIG:TranslationSetting:S3RestApiBaseUrl"];

        if (string.IsNullOrWhiteSpace(serviceBaseUrl) ||
            string.IsNullOrWhiteSpace(s3RestApiBaseUrl))
        {
            return services;
        }

        services.AddTranslationsClient(settings =>
        {
            settings.CacheDuration = 60;
            settings.ServiceBaseUri = new Uri(serviceBaseUrl);
            settings.S3RestApiBaseUri = new Uri(s3RestApiBaseUrl);
            settings.ProjectName = configuration["_N1CONFIG:TranslationSetting:ProjectName"];

            var modules = configuration
                .GetSection("_N1CONFIG:TranslationSetting:Modules")
                .Get<string[]>() ?? Array.Empty<string>();

            var locales = configuration
                .GetSection("_N1CONFIG:TranslationSetting:Locales")
                .Get<string[]>() ?? Array.Empty<string>();

            settings.Modules = new HashSet<string>(modules);
            settings.FetchLocales = new HashSet<string>(locales);
            settings.StorageProviders = new List<IStorageProvider>
            {
                new FileStorageProvider(
                    AppDomain.CurrentDomain.BaseDirectory,
                    "i18n",
                    "Nine1.Livebuy")
            };
        });

        return services;
    }
    catch (Exception ex)
    {
        return services;
    }
}

app.UseMiddleware<TranslationsMiddleware>();

//// 若註冊有異常 先不影響程式啟動
var startupLogger = app.Services
    .GetRequiredService<ILoggerFactory>()
    .CreateLogger("Startup");
try
{
    app.Services.RegisterTranslationsClient();
    startupLogger.LogInformation("RegisterTranslationsClient successfully.");
}
catch
{
    startupLogger.LogError("RegisterTranslationsClient failed.");
}
```
