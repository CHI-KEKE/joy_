```csharp
//// 驗證運費券是否適用於購物車的配送地區
bool isValidShippingCoupon = false;
HashSet<long> validCouponShippingAreas = new HashSet<long>();

if (domesticWeightPricingShippingCoupons.Any())
{
    //// 取得最早到期的運費券
    var coupon = domesticWeightPricingShippingCoupons.OrderBy(x => x.RedeemEndDateTime).First();
    
    //// 解析運費券支援的配送地區ID清單,並處理可能的格式錯誤
    var couponShippingAreas = coupon.DeliveryList?
        .Where(x => string.IsNullOrWhiteSpace(x.ShippingArea) == false)
        .SelectMany(x => x.ShippingArea.Split(','))
        .Select(x => 
        {
            var trimmed = x.Trim();
            return long.TryParse(trimmed, out var areaId) ? (long?)areaId : null;
        })
        .Where(x => x.HasValue)
        .Select(x => x.Value)
        .Distinct()
        .ToHashSet() ?? new HashSet<long>();
    
    //// 取得購物車已選配送方式支援的配送地區ID清單
    var selectedDeliveryList = context.Data?.CheckoutType?.DeliveryTypeList?
        .Select(x => x.ShippingAreaId)
        .ToHashSet() ?? new HashSet<long>();
    
    //// 檢查兩者是否有交集(運費券是否適用於購物車配送地區)
    isValidShippingCoupon = couponShippingAreas.Overlaps(selectedDeliveryList);
    
    //// 儲存有效的配送地區供後續使用
    if (isValidShippingCoupon)
    {
        validCouponShippingAreas = couponShippingAreas;
    }
}
```
