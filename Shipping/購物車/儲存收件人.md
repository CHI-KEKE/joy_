## logic

- AddressBookInsertRequestEntityValidator 驗證 Address.Length < 200
- 呼叫 InsertMemberLocationAsync
  - DB : Webstore , MemberLocation
    - MemberLocation_Cellphone = memberLocation.CellPhone
    - MemberLocation_ZipCode = memberLocation.ZipCode ?? string.Empty
    - MemberLocation_Address = string.IsNullOrWhiteSpace(memberLocation.Floor) ? memberLocation.AddressDetail : string.Concat(memberLocation.AddressDetail, memberLocation.Floor)
    - MemberLocation_County = memberLocation.Country ?? string.Empty
    - MemberLocation_DeliveryTypeDef = memberLocation.DeliveryTypeDef  -- Home / OverSea
    - MemberLocation_ShippingProfileTypeDef = memberLocation.ShippingProfileTypeDef -- Home / OverSea
