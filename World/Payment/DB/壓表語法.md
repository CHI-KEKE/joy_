## 4. 更新 Definition 排序

<br>

```sql
-- 重新設定付款方式的排序
UPDATE Definition 
SET Definition_Sort = CASE Definition_Code
    -- Razer 系列 (10-50)
    WHEN 'CreditCardOnce_Razer' THEN 10          -- Razer 信用卡一次付清
    WHEN 'CreditCardInstallment_Razer' THEN 20   -- Razer 信用卡分期付款
    WHEN 'OnlineBanking_Razer' THEN 30           -- Razer 網路銀行
    WHEN 'TNG_Razer' THEN 40                     -- Razer Touch 'n Go
    WHEN 'Boost_Razer' THEN 50                   -- Razer Boost
    WHEN 'GrabPay_Razer' THEN 60                 -- Razer GrabPay
    
    -- AsiaPay 系列 (70-110)
    WHEN 'CreditCardOnce_AsiaPay' THEN 70         -- AsiaPay 信用卡
    WHEN 'OnlineBanking_AsiaPay' THEN 80          -- AsiaPay 網路銀行
    WHEN 'TNG_AsiaPay' THEN 90                    -- AsiaPay Touch 'n Go
    WHEN 'Boost_AsiaPay' THEN 100                 -- AsiaPay Boost
    WHEN 'GrabPay_AsiaPay' THEN 110               -- AsiaPay GrabPay
    
    -- 其他付款方式 (120-200)
    WHEN 'Atome' THEN 120                         -- Atome
    WHEN 'GlobalPay' THEN 130                     -- Online Payment
    WHEN 'CustomOfflinePayment' THEN 140          -- Custom Offline Payment
    WHEN 'FreeOfCharge' THEN 150                  -- Free Of Charge
    WHEN 'GooglePay' THEN 160                     -- Google Pay
    WHEN 'StoreCredit' THEN 170                   -- 購物金
    
    ELSE Definition_Sort  -- 保持其他記錄的原始排序
END,
Definition_UpdatedDateTime = GETDATE(),
Definition_UpdatedUser = 'SYSTEM_SORT_UPDATE'  -- 請替換為實際的使用者識別
WHERE Definition_TableName = 'PayProfile' 
  AND Definition_ColumnName = 'PayProfile_StatisticsTypeDef'
  AND Definition_Code IN (
    'CreditCardOnce_Razer', 'CreditCardInstallment_Razer', 'OnlineBanking_Razer', 
    'TNG_Razer', 'Boost_Razer', 'GrabPay_Razer',
    'CreditCardOnce_AsiaPay', 'OnlineBanking_AsiaPay', 'TNG_AsiaPay', 
    'Boost_AsiaPay', 'GrabPay_AsiaPay',
    'Atome', 'GlobalPay', 'CustomOfflinePayment', 'FreeOfCharge', 
    'GooglePay', 'StoreCredit'
  );
```
