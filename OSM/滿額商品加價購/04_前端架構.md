## 一、AngularJS 層（舊版選擇活動類型頁面）

### selectActivityForm.controller.ts
**路徑**：`WebSite\WebSite\TypeScripts\Modules\Promotion\selectActivityForm.controller.ts`

#### 關鍵屬性
```typescript
// 是否顯示購物車活動，滿額加價購
public IsEnableCartReachPriceExtraPurchase: boolean = false;

// 購物車活動列表 ViewModel
public CartPromotionViewModel: PromotionCategoryViewModel;
```

#### 初始化流程
```typescript
// 呼叫 getShopStaticSettings 取得 Toggle
this.PromotionEngineService.getShopStaticSettings({
    shopId: this.ShopId,
    keys: ['EnableCartReachPriceExtraPurchase'],
})
.then((shopSettings) => {
    const enableCartReachPriceExtraPurchase = _.find(shopSettings, {
        Key: 'EnableCartReachPriceExtraPurchase',
    });
    if (enableCartReachPriceExtraPurchase) {
        this.IsEnableCartReachPriceExtraPurchase =
            enableCartReachPriceExtraPurchase.Enable;
    }
})
```

#### 建立購物車活動分類
```typescript
// 新增購物車活動分類
this.CartPromotionViewModel = {
    Title: '購物車活動',
    IsActive: false,
    PromotionList: [],
};

if (this.IsEnableCartReachPriceExtraPurchase) {
    this.CartPromotionViewModel.PromotionList.push({
        Title: '滿額加價購',
        TypeDef: 'CartReachPriceExtraPurchase',
        Description: '消費滿指定金額，即可加購商品',
    });
}
```
