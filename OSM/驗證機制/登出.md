/// <summary>
/// 登出
/// </summary>
/// <param name="sessionId">使用者的Session Id</param>
/// <param name="context">IOwinContext</param>
private void Logout(string sessionId, IOwinContext context)
{
    //// 清除使用者資料
    this._cacheProvider.Remove(string.Format(StateMachineKeyTemplates.LoginKey, sessionId));
    this._cacheProvider.Remove(string.Format(StateMachineKeyTemplates.UserDataKey, sessionId));
    this._cacheProvider.Remove(string.Format(StateMachineKeyTemplates.SessionKey, sessionId));
    //// 登出
    HttpContext.Current.Request.Cookies.Remove(".AspNet.Cookies");
    context.Authentication.SignOut();
}
