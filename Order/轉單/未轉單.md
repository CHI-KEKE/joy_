**補建立 Task**

```SQL
USE WebStoreDB
GO
  
DECLARE @TgCode VARCHAR(100)
SET @TgCode = 'TG251030K00010' -- TODO: 置換 TGCode
  
DROP TABLE IF EXISTS #myInvoice;
   
SELECT
    MyInvoice_TradesOrderGroupId INTO #myInvoice
FROM [ERPROLS].ERPDB.dbo.MyInvoice WITH (NOLOCK)
WHERE MyInvoice_TradesOrderGroupCode IN (SELECT
        value
    FROM STRING_SPLIT(@TgCode, ','))
AND MyInvoice_ValidFlag = 1;
   
WITH memberId_cte (MemberId, 首購_Task_Data, MyInvoice_Task_Data, TransferStripePaymentMetadata_Task_Data, TgId/**/)
AS
(SELECT DISTINCT
        TradesOrderGroup_MemberId
       ,CONCAT('{\"ShopId\":', TradesOrderGroup_ShopId, ',\"MemberId\":\"', TradesOrderGroup_MemberId, '\",\"IsFirstPurchase\":false}') AS N'首購 Task_Data'
       ,CONCAT('{\"ShopId\":', TradesOrderGroup_ShopId, ',\"TradesOrderGroupCode\":\"', TradesOrderGroup_Code, '\"}') AS N'MyInvoice Task_Data'
       ,CONCAT('{\"TradesOrderGroupId\":', TradesOrderGroup_Id, '}') AS N'TransferStripePaymentMetadata Task_Data'
       ,TradesOrderGroup_Id
    FROM dbo.TradesOrderGroup WITH (NOLOCK)
    WHERE TradesOrderGroup_Code IN (SELECT
            value
        FROM STRING_SPLIT(@TgCode, ',')))
   
SELECT
    TradesOrderGroup_ShopId
   ,TradesOrderGroup_MemberId
   ,COUNT(TradesOrderGroup_Code) AS N'曾經成立訂單數' -- 首購訂單：1
   ,CASE
        WHEN COUNT(TradesOrderGroup_Code) = 1 THEN 首購_Task_Data
        ELSE NULL
    END AS 首購_UpdateIsFirstPurchase_Task_Data
   ,TgId AS 轉單_TransferOrderToERP_Task_Data
   ,CASE
        WHEN EXISTS (SELECT
                    *
                FROM #myInvoice
                WHERE TgId = MyInvoice_TradesOrderGroupId) THEN NULL
        ELSE MyInvoice_Task_Data
    END AS 發票_GenerateMyInvoiceTask_Task_Data
   ,TransferStripePaymentMetadata_Task_Data
FROM dbo.TradesOrderGroup WITH (NOLOCK)
INNER JOIN memberId_cte
    ON TradesOrderGroup_MemberId = MemberId
GROUP BY TradesOrderGroup_ShopId
        ,TradesOrderGroup_MemberId
        ,首購_Task_Data
        ,MyInvoice_Task_Data
        ,TransferStripePaymentMetadata_Task_Data
        ,TgId
```

USE NMQV2DB
 
DECLARE @tgId BIGINT = 2375608 -- TODO TG Id
DECLARE @tgCode VARCHAR(50) = 'TG251030K00010' -- TODO TG Code
DECLARE @start DATETIME = '2025-10-30 09:20' -- TODO 訂單成立時間
DECLARE @end DATETIME = DATEADD(HOUR, 1, @start)
 
SELECT
    Job_Name
    , Task_Data
    , Task_CreatedDateTime
FROM dbo.Job(NOLOCK)
JOIN dbo.Task(NOLOCK)
    ON Task_ValidFlag  = 1
    AND Task_JobId = Job_Id
    AND Task_UpdatedDateTime >= @start
    AND Task_UpdatedDateTime <= @end
WHERE Job_ValidFlag = 1
    AND Job_Name IN ('TransferOrderToERP','TransferStripePaymentMetadata','GenerateMyInvoiceTask','UpdateIsFirstPurchase')
    AND Task_Data LIKE CONCAT('%', @tgId,'%')
    OR Task_Data LIKE CONCAT('%', @tgCode, '%');
